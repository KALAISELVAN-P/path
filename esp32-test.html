ESP32 Connection Test

This tool helps test your ESP32 connection at 192.168.22.122

Expected ESP32 Endpoints:

  • http://192.168.22.122/ - Main page
  • http://192.168.22.122/status - Status check
  • http://192.168.22.122/api/potholes - Pothole data
  • http://192.168.22.122/info - Device info

Sample ESP32 Arduino Code:

#include <WiFi.h>
#include <WebServer.h>
#include <ArduinoJson.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";

WebServer server(80);

void setup() {
  Serial.begin(115200);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  
  Serial.println("WiFi connected!");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
  
  // Enable CORS for all routes
  server.enableCORS(true);
  
  server.on("/", handleRoot);
  server.on("/status", handleStatus);
  server.on("/api/potholes", handlePotholes);
  server.on("/info", handleInfo);
  
  server.begin();
  Serial.println("HTTP server started");
}

void loop() {
  server.handleClient();
}

void handleRoot() {
  server.send(200, "text/html", "<h1>ESP32 Pothole Detection System</h1>");
}

void handleStatus() {
  server.send(200, "application/json", "{\"status\":\"online\",\"device\":\"ESP32\"}");
}

void handlePotholes() {
  String json = "[{\"id\":1,\"latitude\":11.0168,\"longitude\":76.9558,\"depth\":5.2,\"severity\":\"medium\",\"timestamp\":\"2024-01-01T10:00:00Z\",\"status\":\"pending\"}]";
  server.send(200, "application/json", json);
}

void handleInfo() {
  String json = "{\"device\":\"ESP32\",\"ip\":\"192.168.22.122\",\"status\":\"online\"}";
  server.send(200, "application/json", json);
}
